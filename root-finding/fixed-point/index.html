mathjs = window.math
compiledG = {
  try {
    return mathjs.compile(String(gText).trim().replaceAll("π", "pi"))
  } catch (err) {
    return null
  }
}
g = x => {
  if (!compiledG) return NaN
  try {
    const y = compiledG.evaluate({x})
    return Number(y)
  } catch (err) {
    return NaN
  }
}
fixedPointResult = {
  const rows = []
  const start = Number(x0)
  const tol = Number(tolerance)
  const maxIter = Number(maxIterations)

  if (!compiledG) {
    return {
      ok: false,
      statusType: "bad",
      message: "Could not parse g(x). Try an expression such as cos(x), exp(-x), or sqrt(x + 2).",
      rows
    }
  }

  if (!Number.isFinite(start)) {
    return {
      ok: false,
      statusType: "bad",
      message: "The initial guess x₀ must be a finite number.",
      rows
    }
  }

  let x = start

  for (let i = 0; i < maxIter; i++) {
    const gx = g(x)

    if (!Number.isFinite(gx)) {
      rows.push({
        i,
        x,
        gx,
        error: NaN
      })

      return {
        ok: false,
        statusType: "bad",
        message: `Iteration stopped because g(x_${i}) is not finite.`,
        rows
      }
    }

    const error = Math.abs(gx - x)

    rows.push({
      i,
      x,
      gx,
      error
    })

    if (error < tol) {
      return {
        ok: true,
        statusType: "good",
        message: `Converged after ${i + 1} iteration${i + 1 === 1 ? "" : "s"}. Approximate fixed point: x ≈ ${gx.toPrecision(10)}.`,
        rows
      }
    }

    x = gx
  }

  return {
    ok: true,
    statusType: "warn",
    message: `Reached the maximum number of iterations. Last approximation: x ≈ ${x.toPrecision(10)}.`,
    rows
  }
}
visibleRows = fixedPointResult.rows.slice(
  0,
  Math.min(Number(stepControl), fixedPointResult.rows.length) + 1
)
fixedPointPlotRange = {
  const useAuto = autoRange.includes("Automatically adjust axes")

  if (!useAuto) {
    let lo = Number(xminManual)
    let hi = Number(xmaxManual)

    if (!Number.isFinite(lo) || !Number.isFinite(hi) || lo === hi) {
      lo = Number(x0) - 2
      hi = Number(x0) + 2
    }

    if (lo > hi) {
      const temp = lo
      lo = hi
      hi = temp
    }

    return {lo, hi}
  }

  const values = []

  const start = Number(x0)
  if (Number.isFinite(start)) values.push(start)

  for (const row of visibleRows) {
    if (Number.isFinite(row.x)) values.push(row.x)
    if (Number.isFinite(row.gx)) values.push(row.gx)
  }

  let lo
  let hi

  if (values.length === 0) {
    lo = -2
    hi = 2
  } else {
    lo = Math.min(...values)
    hi = Math.max(...values)

    if (lo === hi) {
      lo = lo - 2
      hi = hi + 2
    }
  }

  const width = hi - lo
  const pad = Math.max(0.45 * width, 1.0)

  return {
    lo: lo - pad,
    hi: hi + pad
  }
}

    const start = Number(x0)

    if (Number.isFinite(start)) {

      cobwebY.push(start)
    }

    for (const row of visibleRows) {

      // Vertical move: from y = x_i to y = g(x_i)
      cobwebX.push(row.x)
      cobwebY.push(row.gx)

      // Horizontal move: from x_i to x_{i+1} along y = g(x_i)
      cobwebX.push(row.gx)
      cobwebY.push(row.gx)
    }

    data.push({
      x: cobwebX,
      y: cobwebY,
      type: "scatter",
      mode: "lines+markers",
      name: "Cobweb path",
      line: {
        color: "#dc2626",
        width: 2
      },
      marker: {
        color: "#dc2626",
        size: 7
      }
    })

    data.push({
      x: visibleRows.map(row => row.x).filter(Number.isFinite),
      y: visibleRows.map(row => row.x).filter(Number.isFinite),
      type: "scatter",
      mode: "markers+text",
      name: "Iterates",

      zeroline: true,
      zerolinecolor: "#94a3b8",
      gridcolor: "#e2e8f0",

}

Explanation

Fixed point iteration is a way to solve an equation after rewriting it in the form

\[ x=g(x). \]

Starting with an initial guess \(x_0\), the method repeatedly applies \(g\):

\[ x_{n+1}=g(x_n). \]

If the sequence settles down to a value \(\alpha\), then that value satisfies

\[ \alpha=g(\alpha). \]

So \(\alpha\) is called a fixed point of \(g\).

What the picture is showing

The plot shows two curves:

  • the function \(y=g(x)\),
  • the line \(y=x\).

The red path shows the iteration. From the current point, the method moves vertically to the curve \(y=g(x)\), then horizontally to the line \(y=x\). Repeating this creates the cobweb pattern.

This visualization is useful because it shows whether the iteration appears to be moving toward a fixed point, moving away from it, or oscillating.

A common condition that helps guarantee local convergence is

\[ |g'(x)|<1 \]

near the fixed point.
