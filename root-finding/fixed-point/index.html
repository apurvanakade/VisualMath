Fixed Point Iteration

No derivatives
Fixed points
Linear convergence
Slow convergence
Rewrites an equation as x = g(x) and iterates x_{n+1} = g(x_n), visualized as a cobweb diagram. 
// VM is the shared utility library loaded from js/utils.js.
VM = window.VM

Fixed point iteration finds a value \(\alpha\) where \(g(\alpha) = \alpha\) by repeatedly applying \(g\): starting from a guess \(x_0\), it computes \(x_1 = g(x_0)\), \(x_2 = g(x_1)\), and so on.

// precompute runs fixed-point iteration and returns all iteration data.
//
// Parameters:
//   gText        — user's iteration function string, e.g. "cos(x)"
//   x0           — initial guess
//   maxIterations — number of steps to run
//
// Returns: { rows, g }
//   rows — array of { i, x, gx, error }, one entry per iteration
//   g    — parsed function, reused for curve sampling in renderOutput
precompute = ({gText, x0, maxIterations}) => {
  const math = window.math

  const g        = VM.makeFunction(math, gText)
  const fallback = x => NaN

  if (!g) return {rows: [], g: fallback}

  const rows  = []
  const start = Number(x0)

  if (!Number.isFinite(start)) return {rows: [], g}

  let x = start
  let i = 0
  while (i < maxIterations) {
    const gx = g(x)

    // Stop if g(x) blows up — the iteration has diverged.
    if (!Number.isFinite(gx)) {
      rows.push({i, x, gx, error: NaN})
      return {rows, g}
    }

    const error = Math.abs(gx - x)
    rows.push({i, x, gx, error})

    x = gx
    i++
  }
  return {rows, g}
}
// renderOutput builds the cobweb chart and iteration table for a given step.
//
// _plotDiv is kept in a closure so Plotly reuses the same DOM node across
// reactive updates — this preserves zoom and pan when the step changes.
renderOutput = {
  const Plotly = window.Plotly
  let _plotDiv  = null
  let _plotDiv2 = null
  let _plotDiv3 = null

  return ({html, tex, result, stepControl, x0}) => {
    // Rows 0..stepControl — what's visible at the current step.
    const visibleRows = result.rows.slice(0, Math.min(Number(stepControl), result.rows.length) + 1)

    // Axis ranges from ALL rows so the axes stay fixed while stepping.
    // The cobweb diagram uses a square plot (same range for x and y).
    const rangeValues = [Number(x0)]
    for (const row of result.rows) VM.pushFiniteValues(rangeValues, [row.x, row.gx])
    const range = VM.paddedRange(rangeValues, {emptyRange: [-2, 2], singularPadding: 2, relativePadding: 0.45, minPadding: 1.0})

    // Sample g across the range for the smooth curve trace.
    const N  = 500
    const xs = Array.from({length: N}, (_, i) => range.lo + (range.hi - range.lo) * i / (N - 1))
    const ys = xs.map(x => result.g(x)).map(y => Number.isFinite(y) ? y : null)

    // --- Plotly traces ---

    const data = [
      // The iteration function g(x)
      {x: xs, y: ys, type: "scatter", mode: "lines", name: "g(x)", showlegend: true, line: {color: "#2563eb", width: 3}},
      // The diagonal y = x — fixed points lie where g(x) crosses this line
      {x: [range.lo, range.hi], y: [range.lo, range.hi], type: "scatter", mode: "lines", name: "y = x", showlegend: true, line: {color: "#111827", width: 2, dash: "dash"}},
      // The x-axis reference
      {x: [range.lo, range.hi], y: [0, 0], type: "scatter", mode: "lines", showlegend: false, line: {color: "#94a3b8", width: 1}}
    ]

    if (visibleRows.length > 0) {
      // Build the cobweb staircase path.
      // From x_i on the diagonal, go vertically to (x_i, g(x_i)) on the curve,
      // then horizontally to (g(x_i), g(x_i)) on the diagonal.
      const cobwebX = []
      const cobwebY = []
      const start   = Number(x0)
      if (Number.isFinite(start)) { cobwebX.push(start); cobwebY.push(start) }
      for (const row of visibleRows) {
        if (!Number.isFinite(row.x) || !Number.isFinite(row.gx)) continue
        cobwebX.push(row.x,  row.gx)
        cobwebY.push(row.gx, row.gx)
      }

      data.push(
        // The staircase path
        {x: cobwebX, y: cobwebY, type: "scatter", mode: "lines+markers", name: "Cobweb path", showlegend: true,
          line: {color: "#dc2626", width: 2}, marker: {color: "#dc2626", size: 7}},
        // Green dots marking each iterate x_i on the diagonal
        {
          x: visibleRows.map(r => r.x).filter(Number.isFinite),
          y: visibleRows.map(r => r.x).filter(Number.isFinite),
          type: "scatter", mode: "markers+text", name: "Iterates", showlegend: true,
          text: visibleRows.filter(r => Number.isFinite(r.x)).map(r => `x_${r.i}`),
          textposition: "top center",
          marker: {color: "#16a34a", size: 9, symbol: "circle", line: {color: "white", width: 1}}
        }
      )
    }

    const layout = {
      // title: {text: "Fixed Point Iteration Cobweb Diagram"},
      margin: {l: 0, r: 0, t: 0, b: 0},
      xaxis: {title: "x", range: [range.lo, range.hi], zeroline: true},
      yaxis: {title: "y", range: [range.lo, range.hi], zeroline: true, zerolinecolor: "#94a3b8", gridcolor: "#e2e8f0", linecolor: "#cbd5e1"},
      legend: {orientation: "h", y: -0.18},
      hovermode: "closest",
      uirevision: "static",  // keeps zoom/pan state across reactive updates
      autosize: true
    }
    const config = {responsive: true, displaylogo: false}

    // First render: create the div. Later renders: update in place (preserves zoom).
    if (!_plotDiv) {
      _plotDiv = document.createElement("div")
      _plotDiv.className = "plotly-box-large"
      Plotly.newPlot(_plotDiv, data, layout, config)
      setTimeout(() => Plotly.Plots.resize(_plotDiv), 0)
    } else {
      Plotly.react(_plotDiv, data, layout, config)
    }

    // --- Sub-chart 1: x_n vs n ---
    const xnData = [{
      x: visibleRows.map(r => r.i),
      y: visibleRows.map(r => Number.isFinite(r.x) ? r.x : null),
      type: "scatter", mode: "markers+lines", name: "xₙ",
      line: {color: "#2563eb"}, marker: {color: "#2563eb", size: 7}
    }]
    const xnLayout = {
      margin: {l: 55, r: 10, t: 30, b: 40},
      xaxis: {title: "n", dtick: 1},
      yaxis: {title: "xₙ"},
      title: {text: "Iterates xₙ", font: {size: 13}},
      uirevision: "iterates",
      autosize: true
    }

    if (!_plotDiv2) {
      _plotDiv2 = document.createElement("div")
      _plotDiv2.className = "plotly-box-small"
      Plotly.newPlot(_plotDiv2, xnData, xnLayout, config)
      setTimeout(() => Plotly.Plots.resize(_plotDiv2), 0)
    } else {
      Plotly.react(_plotDiv2, xnData, xnLayout, config)
    }

    // --- Sub-chart 2: log₁₀|x_n - g(x_n)| vs log₁₀(n) with linear regression ---
    const logRows = visibleRows.filter(r => r.i > 0 && Number.isFinite(r.error) && r.error > 0)
    const logNs   = logRows.map(r => Math.log10(r.i))
    const logYs   = logRows.map(r => Math.log10(r.error))

    const logData = [{
      x: logNs, y: logYs,
      type: "scatter", mode: "markers", name: "log₁₀|xₙ − g(xₙ)|",
      marker: {color: "#dc2626", size: 7}
    }]

    if (logNs.length >= 2) {
      const m   = logNs.length
      const sx  = logNs.reduce((a, b) => a + b, 0)
      const sy  = logYs.reduce((a, b) => a + b, 0)
      const sxy = logNs.reduce((s, x, i) => s + x * logYs[i], 0)
      const sxx = logNs.reduce((s, x) => s + x * x, 0)
      const denom = m * sxx - sx * sx
      if (denom !== 0) {
        const slope     = (m * sxy - sx * sy) / denom
        const intercept = (sy - slope * sx) / m
        const xlo = logNs[0], xhi = logNs[m - 1]
        logData.push({
          x: [xlo, xhi], y: [slope * xlo + intercept, slope * xhi + intercept],
          type: "scatter", mode: "lines",
          name: `slope ≈ ${slope.toFixed(2)}`,
          line: {color: "#9333ea", width: 2, dash: "dash"}
        })
      }
    }

    const logLayout = {
      margin: {l: 60, r: 10, t: 30, b: 55},
      xaxis: {title: "log₁₀(n)"},
      yaxis: {title: "log₁₀|xₙ − g(xₙ)|"},
      title: {text: "Convergence rate", font: {size: 13}},
      legend: {orientation: "h", y: -0.35},
      uirevision: "convergence",
      autosize: true
    }

    if (!_plotDiv3) {
      _plotDiv3 = document.createElement("div")
      _plotDiv3.className = "plotly-box-small"
      Plotly.newPlot(_plotDiv3, logData, logLayout, config)
      setTimeout(() => Plotly.Plots.resize(_plotDiv3), 0)
    } else {
      Plotly.react(_plotDiv3, logData, logLayout, config)
    }

    // Iteration table — shows one row per step
    const table = visibleRows.length === 0
      ? null
      : VM.renderTable({
          html,
          headers: ["n", "x_n", tex`g(x_n) = x_{n+1}`, tex`|x_{n+1} - x_n|`],
          rows: visibleRows.map(row => [
            row.i,
            Number.isFinite(row.x)     ? row.x.toPrecision(10)    : "NaN",
            Number.isFinite(row.gx)    ? row.gx.toPrecision(10)   : "NaN",
            Number.isFinite(row.error) ? row.error.toExponential(4) : "NaN"
          ])
        })

    return html`<div>${_plotDiv}<div class="vm-subcharts-row">${_plotDiv2}${_plotDiv3}</div>${table}</div>`
  }
}

  el.style.width = "100%"
  return el
}

  label: "Initial guess x₀",
  value: 1,
  step: 0.1
})

  const el = Inputs.range([0, maxIterations - 1], {step: 1, value: 0, label: "Step"})
  el.style.width = "100%"
  return el
}

renderOutput({html, tex, result, stepControl, x0})

Explanation

Fixed point iteration is a way to solve an equation after rewriting it in the form

\[ x=g(x). \]

Starting with an initial guess \(x_0\), the method repeatedly applies \(g\):

\[ x_{n+1}=g(x_n). \]

If the sequence settles down to a value \(\alpha\), then that value satisfies

\[ \alpha=g(\alpha). \]

So \(\alpha\) is called a fixed point of \(g\).

What the picture is showing

The plot shows two curves:

  • the function \(y=g(x)\),
  • the line \(y=x\).

The red path shows the iteration. From the current point, the method moves vertically to the curve \(y=g(x)\), then horizontally to the line \(y=x\). Repeating this creates the cobweb pattern.

This visualization is useful because it shows whether the iteration appears to be moving toward a fixed point, moving away from it, or oscillating.

A common condition that helps guarantee local convergence is

\[ |g'(x)|<1 \]

near the fixed point.
