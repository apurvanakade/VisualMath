Newton’s Method

Derivatives
Fast convergence
Quadratic convergence
Root finding
Uses tangent lines and derivative information to find roots of f(x)=0 with fast convergence. 
// VM is the shared utility library loaded from js/make-function.js, js/make-derivative.js, js/padded-range.js, js/render-table.js.
// It provides makeFunction, makeDerivative, paddedRange, renderTable.
VM = window.VM

Newton’s method finds roots of \(f(x) = 0\) by repeatedly linearizing the function: starting from a guess \(x_0\), it draws the tangent line at \((x_n, f(x_n))\) and uses its \(x\)-intercept as the next guess.

// setup computes the data shared by all three plots below: which rows are
// visible at the current step, and the axis ranges / curve samples for the
// main chart.
setup = {
  const visibleRows = result.rows.slice(0, Math.min(Number(stepControl), result.rows.length) + 1)

  // Axis ranges are computed from ALL rows, not just visibleRows,
  // so the axes stay fixed as the user steps through iterations.
  const xValues = [Number(x0)]
  for (const row of result.rows) xValues.push(row.x, row.xNext)
  const xRange = VM.paddedRange(xValues, {emptyRange: [-3, 3], singularPadding: 3, relativePadding: 0.45, minPadding: 1.0})

  // Sample f across the x range for the smooth curve trace.
  const N = 700
  const sampleXs = Array.from({length: N}, (_, i) => xRange.lo + (xRange.hi - xRange.lo) * i / (N - 1))
  const sampleYs = sampleXs.map(x => result.f(x)).map(y => Number.isFinite(y) && Math.abs(y) < 1e8 ? y : null)

  // Collect y values for the y-axis range, including tangent line endpoints.
  // Cap at 1e8 so a near-vertical tangent doesn't blow up the axis range.
  const yValues = [0, ...sampleYs]
  for (const row of result.rows) {
    if (Number.isFinite(row.fx) && Math.abs(row.fx) <= 1e8) yValues.push(row.fx)
    if (Number.isFinite(row.x) && Number.isFinite(row.fx) && Number.isFinite(row.dfx) && Number.isFinite(row.xNext)) {
      const tlo = Math.max(xRange.lo, Math.min(row.x, row.xNext) - 0.75)
      const thi = Math.min(xRange.hi, Math.max(row.x, row.xNext) + 0.75)
      for (const v of [row.fx + row.dfx * (tlo - row.x), row.fx + row.dfx * (thi - row.x)]) {
        if (Number.isFinite(v) && Math.abs(v) <= 1e8) yValues.push(v)
      }
    }
  }
  const yRange = VM.paddedRange(yValues, {emptyRange: [-1, 1], singularPadding: 1, relativePadding: 0.15, minPadding: 0.5})

  return {visibleRows, xRange, yRange, sampleXs, sampleYs}
}
// mainPlot draws f(x), the tangent lines, the Newton points, and the
// tangent x-intercepts.
//
// _plotDiv is kept in a closure so Plotly reuses the same DOM node across
// reactive updates — this preserves zoom and pan when the step changes.
mainPlot = {
  const Plotly = window.Plotly
  let _plotDiv = null

  return ({visibleRows, xRange, yRange, sampleXs, sampleYs}) => {
    const data = [
      // The function curve
      {x: sampleXs, y: sampleYs, type: "scatter", mode: "lines", name: "f(x)", showlegend: true, line: {color: "#2563eb", width: 3}},
      // The x-axis (y = 0 reference line)
      {x: [xRange.lo, xRange.hi], y: [0, 0], type: "scatter", mode: "lines", name: "y = 0", showlegend: true, line: {color: "#111827", width: 2}}
    ]

    // Dotted vertical lines from the x-axis up to each Newton point
    const newtonPoints = visibleRows.filter(r => Number.isFinite(r.x) && Number.isFinite(r.fx))
    for (const r of newtonPoints) {
      data.push({x: [r.x, r.x], y: [0, r.fx], type: "scatter", mode: "lines", showlegend: false, hoverinfo: "skip", line: {color: "#94a3b8", width: 1.5, dash: "dot"}})
    }

    // Tangent lines: previous ones faded, current one solid
    const validTangentRows = visibleRows.filter(r =>
      Number.isFinite(r.x) && Number.isFinite(r.fx) && Number.isFinite(r.dfx) && Number.isFinite(r.xNext)
    )
    const prevTangents    = validTangentRows.slice(0, -1)
    const currentTangent  = validTangentRows.length > 0 ? validTangentRows[validTangentRows.length - 1] : null

    for (const row of prevTangents) {
      const tlo = Math.max(xRange.lo, Math.min(row.x, row.xNext) - 0.75)
      const thi = Math.min(xRange.hi, Math.max(row.x, row.xNext) + 0.75)
      data.push({
        x: [tlo, thi], y: [row.fx + row.dfx * (tlo - row.x), row.fx + row.dfx * (thi - row.x)],
        type: "scatter", mode: "lines", showlegend: false, hoverinfo: "skip",
        line: {color: "rgba(220, 38, 38, 0.18)", width: 1.5}
      })
    }
    if (currentTangent) {
      const row = currentTangent
      const tlo = Math.max(xRange.lo, Math.min(row.x, row.xNext) - 0.75)
      const thi = Math.min(xRange.hi, Math.max(row.x, row.xNext) + 0.75)
      data.push({
        x: [tlo, thi], y: [row.fx + row.dfx * (tlo - row.x), row.fx + row.dfx * (thi - row.x)],
        type: "scatter", mode: "lines", name: "Tangent line", showlegend: true,
        line: {color: "#dc2626", width: 2.5}
      })
    }

    // Red markers at each (x_i, f(x_i)) Newton point
    if (newtonPoints.length > 0) {
      data.push({
        x: newtonPoints.map(r => r.x), y: newtonPoints.map(r => r.fx),
        type: "scatter", mode: "markers+text", name: "Newton points",
        text: newtonPoints.map(r => `x_${r.i}`), textposition: "top center", showlegend: true,
        marker: {color: "#dc2626", size: 10, symbol: "circle", line: {color: "white", width: 1}}
      })
    }

    // Green markers at each tangent x-intercept (x_{i+1})
    const intercepts = visibleRows.filter(r => Number.isFinite(r.xNext)).map(r => ({i: r.i + 1, x: r.xNext}))
    if (intercepts.length > 0) {
      data.push({
        x: intercepts.map(r => r.x), y: intercepts.map(() => 0),
        type: "scatter", mode: "markers+text", name: "Tangent x-intercepts",
        text: intercepts.map(r => `x_${r.i}`), textposition: "bottom center", showlegend: true,
        marker: {color: "#16a34a", size: 10, symbol: "circle", line: {color: "white", width: 1}}
      })
    }

    const layout = {
      // title: {text: "Newton's Method Tangent Visualization"},
      margin: {l: 0, r: 0, t: 0, b: 0},
      xaxis: {title: "x", range: [xRange.lo, xRange.hi], zeroline: true},
      yaxis: {title: "f(x)", range: [yRange.lo, yRange.hi], zeroline: true},
      legend: {orientation: "h", y: -0.18},
      hovermode: "closest",
      uirevision: "static",  // keeps zoom/pan state across reactive updates
      autosize: true
    }
    const config = {responsive: true, displaylogo: false}

    // First render: create the div. Later renders: update in place (preserves zoom).
    if (!_plotDiv) {
      _plotDiv = document.createElement("div")
      _plotDiv.className = "plotly-box-large"
      Plotly.newPlot(_plotDiv, data, layout, config)
      // newPlot fires before the div is in the DOM, so Plotly measures 0 width.
      // A ResizeObserver catches the real size once it's inserted and laid out,
      // and keeps correcting it on any later layout/viewport change.
      new ResizeObserver(() => Plotly.Plots.resize(_plotDiv)).observe(_plotDiv)
    } else {
      Plotly.react(_plotDiv, data, layout, config)
    }
    return _plotDiv
  }
}

// result: runs Newton's method with the current inputs.
//   rows — array of { i, x, fx, dfx, xNext, error }, one entry per iteration
//   f, df — parsed functions, reused for curve sampling in renderOutput
result = {
  const math = window.math

  // Parse f(x) and its symbolic derivative from the user's string.
  // makeFunction / makeDerivative return null if the expression is invalid.
  const f  = VM.makeFunction(math, fText)
  const df = VM.makeDerivative(math, fText)
  const fallback = x => NaN

  if (!f || !df) return {rows: [], f: fallback, df: fallback}

  const rows  = []
  const start = Number(x0)

  if (!Number.isFinite(start)) return {rows: [], f, df}

  let x = start
  let i = 0
  while (i < maxIterations) {
    const fx  = f(x)
    const dfx = df(x)

    // Stop if f or f' blows up — the method has diverged.
    if (!Number.isFinite(fx) || !Number.isFinite(dfx)) {
      rows.push({i, x, fx, dfx, xNext: NaN, error: NaN})
      return {rows, f, df}
    }

    // Stop if the tangent is nearly flat — dividing by dfx would be unstable.
    if (Math.abs(dfx) < 1e-14) {
      rows.push({i, x, fx, dfx, xNext: NaN, error: NaN})
      return {rows, f, df}
    }

    const xNext = x - fx / dfx        // Newton update: x-intercept of the tangent
    const error = Math.abs(xNext - x)
    rows.push({i, x, fx, dfx, xNext, error})

    x = xNext
    i++
  }
  return {rows, f, df}
}

// convergencePlot draws log₁₀|f(xₙ)| vs log₁₀(n) with a linear regression
// line, for the entire run — it doesn't depend on the current step, so it's
// just plotted once per run.
convergencePlot = {
  const Plotly = window.Plotly
  const rows = result.rows

  const logRows = rows.filter(r => r.i > 0 && Number.isFinite(r.fx) && r.fx !== 0)
  const logNs   = logRows.map(r => Math.log10(r.i))
  const logYs   = logRows.map(r => Math.log10(Math.abs(r.fx)))

  const logData = [{
    x: logNs, y: logYs,
    type: "scatter", mode: "markers", name: "log₁₀|f(xₙ)|",
    marker: {color: "#dc2626", size: 7}
  }]

  if (logNs.length >= 2) {
    const m   = logNs.length
    const sx  = logNs.reduce((a, b) => a + b, 0)
    const sy  = logYs.reduce((a, b) => a + b, 0)
    const sxy = logNs.reduce((s, x, i) => s + x * logYs[i], 0)
    const sxx = logNs.reduce((s, x) => s + x * x, 0)
    const denom = m * sxx - sx * sx
    if (denom !== 0) {
      const slope     = (m * sxy - sx * sy) / denom
      const intercept = (sy - slope * sx) / m
      const xlo = logNs[0], xhi = logNs[m - 1]
      logData.push({
        x: [xlo, xhi], y: [slope * xlo + intercept, slope * xhi + intercept],
        type: "scatter", mode: "lines",
        name: `slope ≈ ${slope.toFixed(2)}`,
        line: {color: "#9333ea", width: 2, dash: "dash"}
      })
    }
  }

  const logLayout = {
    margin: {l: 40, r: 10, t: 30, b: 55},
    xaxis: {title: "log₁₀(n)"},
    yaxis: {},
    title: {text: "Convergence rate", font: {size: 13}},
    legend: {orientation: "h", y: -0.35},
    autosize: true
  }
  const config = {responsive: true, displaylogo: false}

  const plotDiv = document.createElement("div")
  plotDiv.className = "plotly-box-small"
  Plotly.newPlot(plotDiv, logData, logLayout, config)
  const observer = new ResizeObserver(() => Plotly.Plots.resize(plotDiv))
  observer.observe(plotDiv)
  invalidation.then(() => observer.disconnect())
  return plotDiv
}

Explanation

Newton’s method is one of the most common methods for solving equations of the form

\[ f(x)=0. \]

Starting from an initial guess \(x_0\), the method produces new approximations using

\[ x_{n+1}=x_n-\frac{f(x_n)}{f'(x_n)}. \]

What the picture is showing

At the current value \(x_n\), Newton’s method draws the tangent line to the graph of \(y=f(x)\).

The tangent line at \(x_n\) is

\[ y=f(x_n)+f'(x_n)(x-x_n). \]

The next approximation \(x_{n+1}\) is where this tangent line crosses the \(x\)-axis.

Setting \(y=0\) gives

\[ 0=f(x_n)+f'(x_n)(x_{n+1}-x_n). \]

Solving for \(x_{n+1}\) gives the Newton update:

\[ x_{n+1}=x_n-\frac{f(x_n)}{f'(x_n)}. \]

Newton’s method can converge very quickly when the starting point is close to the root and the derivative at the root is not zero. But it can also fail if the derivative is too small or if the starting point is not chosen well.
