Newton’s Method

Explore how Newton’s Method uses tangent lines to approximate roots of nonlinear equations. Change the function, choose an initial guess, and step through the iteration one tangent at a time.

Tangent-line method Uses derivative Fast local convergence Interactive Plotly visualization

mathjs = window.math
compiledF = {
  try {
    return mathjs.compile(String(fText).trim().replaceAll("π", "pi"))
  } catch (err) {
    return null
  }
}
compiledDF = {
  try {
    return mathjs.derivative(String(fText).trim().replaceAll("π", "pi"), "x").compile()
  } catch (err) {
    return null
  }
}
f = x => {
  if (!compiledF) return NaN
  try {
    const y = compiledF.evaluate({x})
    return Number(y)
  } catch (err) {
    return NaN
  }
}
df = x => {
  if (!compiledDF) return NaN
  try {
    const y = compiledDF.evaluate({x})
    return Number(y)
  } catch (err) {
    return NaN
  }
}
newtonResult = {
  const rows = []
  const start = Number(x0)
  const tol = Number(tolerance)
  const maxIter = Number(maxIterations)

  if (!compiledF || !compiledDF) {
    return {
      ok: false,
      statusType: "bad",
      message: "Could not parse f(x) or compute its derivative. Try an expression such as x^3 - x - 2 or cos(x) - x.",
      rows
    }
  }

  if (!Number.isFinite(start)) {
    return {
      ok: false,
      statusType: "bad",
      message: "The initial guess x₀ must be a finite number.",
      rows
    }
  }

  let x = start

  for (let i = 0; i < maxIter; i++) {
    const fx = f(x)
    const dfx = df(x)

    if (!Number.isFinite(fx) || !Number.isFinite(dfx)) {
      rows.push({
        i,
        x,
        fx,
        dfx,
        xNext: NaN,
        error: NaN
      })

      return {
        ok: false,
        statusType: "bad",
        message: `Iteration stopped because f(x_$${i}) or f'(x_$${i}) is not finite.`,
        rows
      }
    }

    if (Math.abs(dfx) < 1e-14) {
      rows.push({
        i,
        x,
        fx,
        dfx,
        xNext: NaN,
        error: NaN
      })

      return {
        ok: false,
        statusType: "bad",
        message: `Iteration stopped because f'(x_${i}) is too close to zero.`,
        rows
      }
    }

    const xNext = x - fx / dfx
    const error = Math.abs(xNext - x)

    rows.push({
      i,
      x,
      fx,
      dfx,
      xNext,
      error
    })

    if (Math.abs(fx) < tol || error < tol) {
      return {
        ok: true,
        statusType: "good",
        message: `Converged after $${i + 1} iteration$${i + 1 === 1 ? "" : "s"}. Approximate root: x ≈ ${xNext.toPrecision(10)}.`,
        rows
      }
    }

    x = xNext
  }

  return {
    ok: true,
    statusType: "warn",
    message: `Reached the maximum number of iterations. Last approximation: x ≈ ${x.toPrecision(10)}.`,
    rows
  }
}
visibleRows = newtonResult.rows.slice(
  0,
  Math.min(Number(stepControl), newtonResult.rows.length) + 1
)
visibleNewtonPoints = {
  const pts = []

  for (const row of visibleRows) {
    if (Number.isFinite(row.x) && Number.isFinite(row.fx)) {
      pts.push({
        i: row.i,
        x: row.x,
        fx: row.fx,
        dfx: row.dfx,
        xNext: row.xNext
      })
    }
  }

  return pts
}
visibleNewtonIntercepts = {
  const pts = []

  for (const row of visibleRows) {
    if (Number.isFinite(row.xNext)) {
      pts.push({
        i: row.i + 1,
        x: row.xNext,
        y: 0
      })
    }
  }

  return pts
}
newtonXRange = {
  const useAuto = autoRange.includes("Automatically adjust axes")

  if (!useAuto) {
    let lo = Number(xminManual)
    let hi = Number(xmaxManual)

    if (!Number.isFinite(lo) || !Number.isFinite(hi) || lo === hi) {
      lo = Number(x0) - 3
      hi = Number(x0) + 3
    }

    if (lo > hi) {
      const temp = lo
      lo = hi
      hi = temp
    }

    return {lo, hi}
  }

  const values = []

  const start = Number(x0)
  if (Number.isFinite(start)) values.push(start)

  for (const row of visibleRows) {
    if (Number.isFinite(row.x)) values.push(row.x)
    if (Number.isFinite(row.xNext)) values.push(row.xNext)
  }

  let lo
  let hi

  if (values.length === 0) {
    lo = -3
    hi = 3
  } else {
    lo = Math.min(...values)
    hi = Math.max(...values)

    if (lo === hi) {
      lo = lo - 3
      hi = hi + 3
    }
  }

  const width = hi - lo
  const pad = Math.max(0.45 * width, 1.0)

  return {
    lo: lo - pad,
    hi: hi + pad
  }
}
newtonYRange = {
  const lo = newtonXRange.lo
  const hi = newtonXRange.hi

  const values = [0]

  const n = 400
  for (let i = 0; i < n; i++) {
    const x = lo + (hi - lo) * i / (n - 1)
    const y = f(x)

    if (Number.isFinite(y) && Math.abs(y) < 1e8) {
      values.push(y)
    }
  }

  for (const row of visibleRows) {
    if (Number.isFinite(row.fx) && Math.abs(row.fx) < 1e8) {
      values.push(row.fx)
    }

    if (
      Number.isFinite(row.x) &&
      Number.isFinite(row.fx) &&
      Number.isFinite(row.dfx) &&
      Number.isFinite(row.xNext)
    ) {
      const tlo = Math.max(lo, Math.min(row.x, row.xNext) - 0.75)
      const thi = Math.min(hi, Math.max(row.x, row.xNext) + 0.75)

      const y1 = row.fx + row.dfx * (tlo - row.x)
      const y2 = row.fx + row.dfx * (thi - row.x)

      if (Number.isFinite(y1) && Math.abs(y1) < 1e8) values.push(y1)
      if (Number.isFinite(y2) && Math.abs(y2) < 1e8) values.push(y2)
    }
  }

  let ymin = Math.min(...values)
  let ymax = Math.max(...values)

  if (ymin === ymax) {
    ymin = ymin - 1
    ymax = ymax + 1
  }

  const height = ymax - ymin
  const pad = Math.max(0.15 * height, 0.5)

  return {
    lo: ymin - pad,
    hi: ymax + pad
  }
}
newtonPlot = {
  const div = html`<div class="plotly-box-large"></div>`

  const xlo = newtonXRange.lo
  const xhi = newtonXRange.hi
  const ylo = newtonYRange.lo
  const yhi = newtonYRange.hi

  const n = 700
  const xs = Array.from({length: n}, (_, i) => xlo + (xhi - xlo) * i / (n - 1))
  const ys = xs.map(x => f(x)).map(y => Number.isFinite(y) && Math.abs(y) < 1e8 ? y : null)

  const data = [
    {
      x: xs,
      y: ys,
      type: "scatter",
      mode: "lines",
      name: "f(x)",
      line: {
        color: "#2563eb",
        width: 3
      }
    },
    {
      x: [xlo, xhi],
      y: [0, 0],
      type: "scatter",
      mode: "lines",
      name: "y = 0",
      line: {
        color: "#111827",
        width: 2
      }
    }
  ]

  // Gray dotted vertical segments from (x_i, 0) to (x_i, f(x_i)).
  for (const pt of visibleNewtonPoints) {
    data.push({
      x: [pt.x, pt.x],
      y: [0, pt.fx],
      type: "scatter",
      mode: "lines",
      name: "Vertical guides",
      showlegend: false,
      line: {
        color: "#94a3b8",
        width: 1.5,
        dash: "dot"
      },
      hoverinfo: "skip"
    })
  }

  // Red dashed tangent lines for every completed Newton point.
  let tangentLegendShown = false

  for (const row of visibleRows) {
    if (
      !Number.isFinite(row.x) ||
      !Number.isFinite(row.fx) ||
      !Number.isFinite(row.dfx) ||
      !Number.isFinite(row.xNext)
    ) {
      continue
    }

    const tlo = Math.max(xlo, Math.min(row.x, row.xNext) - 0.75)
    const thi = Math.min(xhi, Math.max(row.x, row.xNext) + 0.75)

    const tangentXs = [tlo, thi]
    const tangentYs = tangentXs.map(x => row.fx + row.dfx * (x - row.x))

    data.push({
      x: tangentXs,
      y: tangentYs,
      type: "scatter",
      mode: "lines",
      name: tangentLegendShown ? "Tangent lines" : "Tangent lines",
      showlegend: !tangentLegendShown,
      line: {
        color: "#dc2626",
        width: 2.5,
        dash: "dash"
      }
    })

    tangentLegendShown = true
  }

  // Red Newton points: (x_i, f(x_i)).
  if (visibleNewtonPoints.length > 0) {
    data.push({
      x: visibleNewtonPoints.map(pt => pt.x),
      y: visibleNewtonPoints.map(pt => pt.fx),
      type: "scatter",
      mode: "markers+text",
      name: "Newton points",
      text: visibleNewtonPoints.map(pt => `x_${pt.i}`),
      textposition: "top center",
      marker: {
        color: "#dc2626",
        size: 10,
        line: {
          color: "white",
          width: 1
        }
      }
    })
  }

  // Green tangent x-intercepts: (x_{i+1}, 0).
  if (visibleNewtonIntercepts.length > 0) {
    data.push({
      x: visibleNewtonIntercepts.map(pt => pt.x),
      y: visibleNewtonIntercepts.map(pt => pt.y),
      type: "scatter",
      mode: "markers+text",
      name: "Tangent x-intercepts",
      text: visibleNewtonIntercepts.map(pt => `x_${pt.i}`),
      textposition: "bottom center",
      marker: {
        color: "#16a34a",
        size: 10,
        symbol: "circle",
        line: {
          color: "white",
          width: 1
        }
      }
    })
  }

  const layout = {
    title: {
      text: "Newton's Method Tangent Visualization"
    },
    margin: {
      l: 65,
      r: 30,
      t: 60,
      b: 55
    },
    xaxis: {
      title: "x",
      range: [xlo, xhi],
      zeroline: true
    },
    yaxis: {
      title: "f(x)",
      range: [ylo, yhi],
      zeroline: true
    },
    legend: {
      orientation: "h",
      y: -0.18
    },
    hovermode: "closest"
  }

  Plotly.newPlot(div, data, layout, {
    responsive: true,
    displaylogo: false
  })

  return div
}

Explanation

Newton’s method is one of the most common methods for solving equations of the form

\[ f(x)=0. \]

Starting from an initial guess \(x_0\), the method produces new approximations using

\[ x_{n+1}=x_n-\frac{f(x_n)}{f'(x_n)}. \]

What the picture is showing

At the current value \(x_n\), Newton’s method draws the tangent line to the graph of \(y=f(x)\).

The tangent line at \(x_n\) is

\[ y=f(x_n)+f'(x_n)(x-x_n). \]

The next approximation \(x_{n+1}\) is where this tangent line crosses the \(x\)-axis.

Setting \(y=0\) gives

\[ 0=f(x_n)+f'(x_n)(x_{n+1}-x_n). \]

Solving for \(x_{n+1}\) gives the Newton update:

\[ x_{n+1}=x_n-\frac{f(x_n)}{f'(x_n)}. \]

Newton’s method can converge very quickly when the starting point is close to the root and the derivative at the root is not zero. But it can also fail if the derivative is too small or if the starting point is not chosen well.
