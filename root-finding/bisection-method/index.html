Bisection Method

Bracketing
No derivatives
Guaranteed convergence
Root finding
Finds a root by repeatedly halving an interval where f changes sign. Reliable and guaranteed to converge. 
// VM is the shared utility library loaded from js/make-function.js, js/make-derivative.js, js/padded-range.js, js/render-table.js.
VM = window.VM

The bisection method finds a root by repeatedly halving an interval \([a, b]\) where \(f\) changes sign, guaranteeing that each half still contains a root.

// setup computes the data shared by all three plots below: which rows are
// visible at the current step, and the axis ranges / curve samples for the
// main chart.
setup = {
  const visibleRows = result.rows.slice(0, Math.min(Number(stepControl), result.rows.length) + 1)
  const currentRow  = visibleRows.length === 0 ? null : visibleRows[visibleRows.length - 1]

  // Axis ranges from ALL rows so the axes stay fixed while stepping.
  const xValues = [Number(a0), Number(b0)]
  for (const row of result.rows) xValues.push(row.a, row.b, row.m)
  const xRange = VM.paddedRange(xValues, {emptyRange: [-5, 5], singularPadding: 2, relativePadding: 0.25, minPadding: 0.5})

  // Sample f across the x range for the smooth curve trace.
  const N = 700
  const sampleXs = Array.from({length: N}, (_, i) => xRange.lo + (xRange.hi - xRange.lo) * i / (N - 1))
  const sampleYs = sampleXs.map(x => result.f(x)).map(y => Number.isFinite(y) && Math.abs(y) < 1e8 ? y : null)

  // Cap row values at 1e8 so a near-singularity doesn't blow up the axis range.
  const yValues = [0, ...sampleYs]
  for (const row of result.rows) {
    for (const v of [row.fa, row.fb, row.fm]) {
      if (Number.isFinite(v) && Math.abs(v) <= 1e8) yValues.push(v)
    }
  }
  const yRange = VM.paddedRange(yValues, {emptyRange: [-1, 1], singularPadding: 1, relativePadding: 0.15, minPadding: 0.5})

  return {visibleRows, currentRow, xRange, yRange, sampleXs, sampleYs}
}

  let _plotDiv = null

  return ({visibleRows, currentRow, xRange, yRange, sampleXs, sampleYs}) => {

    ]

    // Previous midpoints — all steps before the current one
    const prevRows = visibleRows.slice(0, -1).filter(r => Number.isFinite(r.m) && Number.isFinite(r.fm))
    if (prevRows.length > 0) {
      data.push({
        x: prevRows.map(r => r.m), y: prevRows.map(r => r.fm),

        marker: {color: "#f97316", size: 8, symbol: "circle", line: {color: "white", width: 1}}
      })
    }

    if (currentRow) {
      const row = currentRow
      data.push(
        // Endpoints a and b
        {x: [row.a, row.b], y: [row.fa, row.fb], type: "scatter", mode: "markers+text", name: "Endpoints",
          text: ["a", "b"], textposition: "top center", showlegend: true,
          marker: {color: "#2563eb", size: 11, symbol: "circle", line: {color: "white", width: 1}}},
        // Current midpoint m
        {x: [row.m], y: [row.fm], type: "scatter", mode: "markers+text", name: "Midpoint",
          text: [`m${row.n}`], textposition: "top center", showlegend: true,
          marker: {color: "#dc2626", size: 13, symbol: "circle", line: {color: "white", width: 1}}},
        // Dotted verticals from x-axis to each marker
        {x: [row.a, row.a], y: [0, row.fa], type: "scatter", mode: "lines", showlegend: false, hoverinfo: "skip", line: {color: "#94a3b8", width: 1.5, dash: "dot"}},
        {x: [row.b, row.b], y: [0, row.fb], type: "scatter", mode: "lines", showlegend: false, hoverinfo: "skip", line: {color: "#94a3b8", width: 1.5, dash: "dot"}},
        {x: [row.m, row.m], y: [0, row.fm], type: "scatter", mode: "lines", showlegend: false, hoverinfo: "skip", line: {color: "#dc2626", width: 1.5, dash: "dot"}}
      )
    }

    // Shaded rectangle showing the current bracket [a, b]
    const shapes = currentRow ? [{
      type: "rect", xref: "x", yref: "paper",
      x0: currentRow.a, x1: currentRow.b, y0: 0, y1: 1,
      fillcolor: "rgba(37, 99, 235, 0.14)", line: {width: 0}
    }] : []

    const layout = {
      margin: {l: 0, r: 0, t: 0, b: 0},
      xaxis: {title: "x", range: [xRange.lo, xRange.hi], zeroline: true},
      yaxis: {title: "f(x)", range: [yRange.lo, yRange.hi], zeroline: true},
      shapes,
      legend: {orientation: "h", y: -0.18},
      hovermode: "closest",
      uirevision: "static",  // keeps zoom/pan state across reactive updates
      autosize: true
    }
    const config = {responsive: true, displaylogo: false}

    // First render: create the div. Later renders: update in place (preserves zoom).
    if (!_plotDiv) {
      _plotDiv = document.createElement("div")
      _plotDiv.className = "plotly-box-large"
      Plotly.newPlot(_plotDiv, data, layout, config)
      new ResizeObserver(() => Plotly.Plots.resize(_plotDiv)).observe(_plotDiv)
    } else {
      Plotly.react(_plotDiv, data, layout, config)
    }
    return _plotDiv
  }
}

// result: runs bisection with the current inputs.
//   rows — array of { n, a, b, m, fa, fb, fm, width }, one entry per iteration
//   f    — parsed function, reused for curve sampling in renderOutput
result = {
  const math = window.math

  const f        = VM.makeFunction(math, fText)
  const fallback = x => NaN

  if (!f) return {rows: [], f: fallback}

  let a = Number(a0)
  let b = Number(b0)

  if (!Number.isFinite(a) || !Number.isFinite(b) || a === b) return {rows: [], f}
  if (a > b) { const tmp = a; a = b; b = tmp }   // ensure a < b

  const fa = f(a)
  const fb = f(b)

  if (!Number.isFinite(fa) || !Number.isFinite(fb)) return {rows: [], f}

  // Bisection requires a sign change across [a, b].
  // If an endpoint is already a root, return it immediately.
  if (fa === 0) return {rows: [{n: 0, a, b, m: a, fa, fb, fm: fa, width: b - a}], f}
  if (fb === 0) return {rows: [{n: 0, a, b, m: b, fa, fb, fm: fb, width: b - a}], f}
  if (fa * fb > 0) return {rows: [], f}   // no sign change — can't bracket a root

  const rows = []
  let curA = a, curB = b, curFa = fa, curFb = fb
  let n = 0
  while (n < maxIterations) {
    const m  = 0.5 * (curA + curB)   // midpoint of the current bracket
    const fm = f(m)
    rows.push({n, a: curA, b: curB, m, fa: curFa, fb: curFb, fm, width: curB - curA})

    if (!Number.isFinite(fm)) return {rows, f}

    // Keep the half that still contains a sign change
    if (curFa * fm < 0) { curB = m; curFb = fm } else { curA = m; curFa = fm }
    n++
  }
  return {rows, f}
}

// convergencePlot draws log₁₀|f(mₙ)| vs log₁₀(n) with a linear regression
// line, for the entire run — it doesn't depend on the current step, so it's
// just plotted once per run.
convergencePlot = {
  const Plotly = window.Plotly
  const rows = result.rows

  const logRows = rows.filter(r => r.n > 0 && Number.isFinite(r.fm) && r.fm !== 0)
  const logNs   = logRows.map(r => Math.log10(r.n))
  const logYs   = logRows.map(r => Math.log10(Math.abs(r.fm)))

  const logData = [{
    x: logNs, y: logYs,
    type: "scatter", mode: "markers", name: "log₁₀|f(mₙ)|",
    marker: {color: "#dc2626", size: 7}
  }]

  if (logNs.length >= 2) {
    const m   = logNs.length
    const sx  = logNs.reduce((a, b) => a + b, 0)
    const sy  = logYs.reduce((a, b) => a + b, 0)
    const sxy = logNs.reduce((s, x, i) => s + x * logYs[i], 0)
    const sxx = logNs.reduce((s, x) => s + x * x, 0)
    const denom = m * sxx - sx * sx
    if (denom !== 0) {
      const slope     = (m * sxy - sx * sy) / denom
      const intercept = (sy - slope * sx) / m
      const xlo = logNs[0], xhi = logNs[m - 1]
      logData.push({
        x: [xlo, xhi], y: [slope * xlo + intercept, slope * xhi + intercept],
        type: "scatter", mode: "lines",
        name: `slope ≈ ${slope.toFixed(2)}`,
        line: {color: "#9333ea", width: 2, dash: "dash"}
      })
    }
  }

  const logLayout = {
    margin: {l: 40, r: 10, t: 30, b: 55},
    xaxis: {title: "log₁₀(n)"},
    yaxis: {},
    title: {text: "Convergence rate", font: {size: 13}},
    legend: {orientation: "h", y: -0.35},
    autosize: true
  }
  const config = {responsive: true, displaylogo: false}

  const plotDiv = document.createElement("div")
  plotDiv.className = "plotly-box-small"
  Plotly.newPlot(plotDiv, logData, logLayout, config)
  const observer = new ResizeObserver(() => Plotly.Plots.resize(plotDiv))
  observer.observe(plotDiv)
  invalidation.then(() => observer.disconnect())
  return plotDiv
}

Explanation

At each step, the bisection method computes the midpoint

\[ m=\frac{a+b}{2}. \]

Then it keeps the half of the interval where the sign change still occurs. Repeating this process makes the interval smaller and smaller around the root.

What the picture is showing

The blue curve is the graph of \(f(x)\). The black horizontal line is \(y=0\).

The shaded blue region is the current interval \([a,b]\). The blue points show the endpoint values \((a,f(a))\) and \((b,f(b))\).

The red point is the midpoint \((m,f(m))\). Each step halves the interval and keeps the side that still contains a sign change.
