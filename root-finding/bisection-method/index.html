  const div = html`

mathjs = window.math
cleanExpression = expr => {
  return String(expr)
    .trim()
    .replaceAll("π", "pi")
}
compiledF = {
  try {
    return mathjs.compile(cleanExpression(fText))
  } catch (err) {
    return null
  }
}
f = x => {
  if (!compiledF) return NaN

  try {
    const y = compiledF.evaluate({x})
    return Number(y)
  } catch (err) {
    return NaN
  }
}
bisectionResult = {
  let a = Number(a0)
  let b = Number(b0)
  const tol = Number(tolerance)
  const maxIter = Number(maxIterations)

  if (!compiledF) {
    return {
      valid: false,
      statusType: "bad",
      message: "Could not parse the function. Try x^3 - x - 2, cos(x) - x, or exp(-x) - x.",
      rows: []
    }
  }

  if (!Number.isFinite(a) || !Number.isFinite(b)) {
    return {
      valid: false,
      statusType: "bad",
      message: "The endpoints must be finite numbers.",
      rows: []
    }
  }

  if (a === b) {
    return {
      valid: false,
      statusType: "bad",
      message: "The endpoints a and b must be different.",
      rows: []
    }
  }

  if (a > b) {
    const temp = a
    a = b
    b = temp
  }

  let fa = f(a)
  let fb = f(b)

  if (!Number.isFinite(fa) || !Number.isFinite(fb)) {
    return {
      valid: false,
      statusType: "bad",
      message: "Could not evaluate f(a) or f(b). Check the function and interval.",
      rows: []
    }
  }

  if (fa === 0) {
    return {
      valid: true,
      statusType: "good",
      message: `The left endpoint a = ${a} is already a root.`,
      rows: [{n: 0, a, b, m: a, fa, fb, fm: fa, width: b - a}]
    }
  }

  if (fb === 0) {
    return {
      valid: true,
      statusType: "good",
      message: `The right endpoint b = ${b} is already a root.`,
      rows: [{n: 0, a, b, m: b, fa, fb, fm: fb, width: b - a}]
    }
  }

  if (fa * fb > 0) {
    return {
      valid: false,
      statusType: "bad",
      message: `Bisection requires f(a) and f(b) to have opposite signs. Here f(a) = ${fa.toPrecision(6)} and f(b) = ${fb.toPrecision(6)}.`,
      rows: []
    }
  }

  const rows = []

  for (let n = 0; n <= maxIter; n++) {
    const m = 0.5 * (a + b)
    const fm = f(m)

    rows.push({
      n,
      a,
      b,
      m,
      fa,
      fb,
      fm,
      width: b - a
    })

    if (!Number.isFinite(fm)) {
      return {
        valid: false,
        statusType: "bad",
        message: `Stopped because f(m) is not finite at step ${n}.`,
        rows
      }
    }

    if (Math.abs(fm) < tol || Math.abs(b - a) < tol) {
      const last = rows[rows.length - 1]

      return {
        valid: true,
        statusType: "good",
        message: `Converged after ${n + 1} iteration${n + 1 === 1 ? "" : "s"}. Approximate root: m ≈ ${last.m.toPrecision(10)}.`,
        rows
      }
    }

    if (fa * fm < 0) {
      b = m
      fb = fm
    } else {
      a = m
      fa = fm
    }
  }

  const last = rows[rows.length - 1]

  return {
    valid: true,
    statusType: "warn",
    message: `Reached the maximum number of iterations. Last midpoint: m ≈ ${last.m.toPrecision(10)}, interval width = ${last.width.toExponential(3)}.`,
    rows
  }
}
visibleRows = bisectionResult.rows.slice(
  0,
  Math.min(Number(stepControl), bisectionResult.rows.length - 1) + 1
)
currentBisectionRow = {
  if (visibleRows.length === 0) return null
  return visibleRows[visibleRows.length - 1]
}
bisectionXRange = {
  const useAuto = autoRange.includes("Automatically adjust axes")

  if (!useAuto) {
    let lo = Number(xminManual)
    let hi = Number(xmaxManual)

    if (!Number.isFinite(lo) || !Number.isFinite(hi) || lo === hi) {
      lo = Number(a0) - 1
      hi = Number(b0) + 1
    }

    if (lo > hi) {
      const temp = lo
      lo = hi
      hi = temp
    }

    return {lo, hi}
  }

  const values = []

  const aStart = Number(a0)
  const bStart = Number(b0)

  if (Number.isFinite(aStart)) values.push(aStart)
  if (Number.isFinite(bStart)) values.push(bStart)

  for (const row of visibleRows) {
    if (Number.isFinite(row.a)) values.push(row.a)
    if (Number.isFinite(row.b)) values.push(row.b)
    if (Number.isFinite(row.m)) values.push(row.m)
  }

  let lo
  let hi

  if (values.length === 0) {
    lo = -5
    hi = 5
  } else {
    lo = Math.min(...values)
    hi = Math.max(...values)

    if (lo === hi) {
      lo = lo - 2
      hi = hi + 2
    }
  }

  const width = hi - lo
  const pad = Math.max(0.25 * width, 0.5)

  return {
    lo: lo - pad,
    hi: hi + pad
  }
}
bisectionYRange = {
  const lo = bisectionXRange.lo
  const hi = bisectionXRange.hi
  const values = [0]

  const n = 500

  for (let i = 0; i < n; i++) {
    const x = lo + (hi - lo) * i / (n - 1)
    const y = f(x)

    if (Number.isFinite(y) && Math.abs(y) < 1e8) {
      values.push(y)
    }
  }

  for (const row of visibleRows) {
    if (Number.isFinite(row.fa) && Math.abs(row.fa) < 1e8) values.push(row.fa)
    if (Number.isFinite(row.fb) && Math.abs(row.fb) < 1e8) values.push(row.fb)
    if (Number.isFinite(row.fm) && Math.abs(row.fm) < 1e8) values.push(row.fm)
  }

  let ymin = Math.min(...values)
  let ymax = Math.max(...values)

  if (ymin === ymax) {
    ymin = ymin - 1
    ymax = ymax + 1
  }

  const height = ymax - ymin
  const pad = Math.max(0.15 * height, 0.5)

  return {
    lo: ymin - pad,
    hi: ymax + pad
  }
}

Explanation

At each step, the bisection method computes the midpoint

\[ m=\frac{a+b}{2}. \]

Then it keeps the half of the interval where the sign change still occurs. Repeating this process makes the interval smaller and smaller around the root.

What the picture is showing

The blue curve is the graph of (f(x)). The black horizontal line is (y=0).

The shaded blue region is the current interval ([a,b]). The blue points show the endpoint values ((a,f(a))) and ((b,f(b))).

The red point is the midpoint ((m,f(m))). Each time you click Next iteration, the method halves the interval and keeps the side that still contains a sign change.

If Automatically adjust axes is selected, the plot range updates dynamically when you change the starting interval.
